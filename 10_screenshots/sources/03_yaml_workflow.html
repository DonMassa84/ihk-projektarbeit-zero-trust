# Zero-Trust RBAC – Automatisierte Rechtevergabe
name: Role Request Workflow
on:
  issues:
    types: [opened, labeled]
  workflow_dispatch:
    inputs:
      role:
        description: Rolle
        required: true

jobs:
  validate:
    runs-on: ubuntu-latest
    steps:
      - name: Check required fields
        run: python scripts/validate_request.py
      - name: Policy compliance check
        run: python scripts/policy_check.py

  approve:
    needs: validate
    runs-on: ubuntu-latest
    steps:
      - name: Request approval from manager
        uses: ./.github/actions/approval-request
      - name: Wait for approval
        run: python scripts/wait_for_approval.py

  provision:
    needs: approve
    runs-on: ubuntu-latest
    steps:
      - name: Assign GitHub Team
        run: python scripts/assign_team.py --role \${{ inputs.role }}
      - name: Create Audit Log Entry
        run: python scripts/audit_log.py --action provisioned
